DROP TABLE IF EXISTS VENDAS;

CREATE TABLE VENDAS (
ID_TEMPO INTEGER ,
ID_LOJA INTEGER,
ID_PRODUTO INTEGER,
QUANTIDADE NUMERIC(18,2),
FATURAMENTO NUMERIC(18,2),
PRIMARY KEY (ID_TEMPO,ID_LOJA,ID_PRODUTO),
FOREIGN KEY (ID_TEMPO) REFERENCES TEMPO(ID),
FOREIGN KEY (ID_PRODUTO) REFERENCES PRODUTO(ID),
FOREIGN KEY (ID_LOJA) REFERENCES LOJA(ID)
);
